CREATE DEFINER=`root`@`localhost` PROCEDURE `obtenerNombreFamiliares`(
	IN `nom_hijo` VARCHAR(255)
)
LANGUAGE SQL
NOT DETERMINISTIC
CONTAINS SQL
SQL SECURITY DEFINER
COMMENT ''
BEGIN
    SELECT e.NOMBRE, p.NOMBRESP, p.NOMBRESM FROM estudiante AS e 
	 INNER JOIN padres AS p ON e.IDESTUDIANTE=p.IDPADRES
    WHERE Nombre= nom_hijo;
END
