SELECT e.NOMBRE AS estudiante, p.NOMBRESP, p.ACTIVOP  FROM estudiante AS e
INNER JOIN padres AS p ON p.IDPADRES=e.IDESTUDIANTE
WHERE p.ACTIVOP="no"
#2
SELECT e.NOMBRE AS estudiante, p.NOMBRESM, p.ACTIVOM  FROM estudiante AS e
INNER JOIN padres AS p ON p.IDPADRES=e.IDESTUDIANTE
WHERE p.ACTIVOM="no"
#3
SELECT d.IDDOCENTE, d.NOMBRED, d.APELLIDOD, d.CEDULAD, c.CURSOS, c.PARALELO, p.ANO, COUNT(m.IDDECURSO)AS n_Estudiantes FROM docente AS d
INNER JOIN curso AS c ON d.IDDOCENTE=c.ID_DE_DOCENTE_TUTOR
INNER JOIN matricula AS m ON m.IDDECURSO= c.IDCURSO
INNER JOIN peridodoacademico AS p ON p.IDPERIODO=m.IDDEPERIODOACADEMICO
INNER JOIN estudiante AS e ON e.IDESTUDIANTE=m.IDDELESTUDIANTE
GROUP BY m.IDDECURSO
#4
SELECT pA.IDPERIODO, pA.ANO,
COUNT(CASE  
	when cA.NOTA_FINAL>=15 then"aprovado"
END)AS aprovado,
COUNT(CASE  
	when cA.NOTA_FINAL<15 then"reprovado"
END)AS reprovado 

FROM calificaciones AS cA
INNER JOIN matricula AS m ON cA.IDCALIFICACIONES=m.IDMATRICULA
INNER JOIN peridodoacademico AS pA ON pA.IDPERIODO=m.IDDEPERIODOACADEMICO
GROUP BY pA.ANO
